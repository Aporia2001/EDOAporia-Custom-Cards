Bienvenido a EDOAporia-Custom-Cards.
Para usar mis cartas custom en EDOPro, cree un archivo "user_configs.json" dentro de la carpeta "config", y pegue lo siguiente:
{
	"repos": [
		{
			"url": "https://github.com/Aporia2001/EDOAporia-Custom-Cards",
			"repo_name": "EDOAporia-Custom-Cards",
			"repo_path": "./repositories/EDOAporia-Custom-Cards",
			"has_core": true,
			"core_path": "bin",
			"data_path": "",
			"script_path": "script",
			"should_update": true,
			"should_read": true
		}
	]
}
